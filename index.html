//Template function Unit(m,x,y) { this.mass = m; this.position = createVector(x,y); this.velocity = createVector(0,0); this.acceleration = createVector(0,0); } // Newton's 2nd law: F = M * A // or A = F / M Unit.prototype.applyForce = function(force) { var f = p5.Vector.div(force,this.mass); this.acceleration.add(f); }; Unit.prototype.update = function() { // Velocity changes according to acceleration this.velocity.add(this.acceleration); // position changes by velocity this.position.add(this.velocity); // We must clear acceleration each frame this.acceleration.mult(0); }; Unit.prototype.display = function() { stroke(0); strokeWeight(2); fill(255,127); ellipse(this.position.x,this.position.y,this.mass*16,this.mass*16); }; // Bounce off bottom of window Unit.prototype.checkEdges = function() { if (this.position.y > (height - this.mass*8)) { // A little dampening when hitting the bottom this.velocity.y *= -0.9; this.position.y = (height - this.mass*8); } };
